#version 330 core
in vec3 position;
in vec3 normal;

out vec3 Normal;
out vec3 Position;

uniform mat4 M;
uniform mat4 V;
uniform mat4 P;

void main()
{
    Normal = mat3(transpose(inverse(M))) * normal;
    Position = vec3(M * vec4(position, 1.0));
    gl_Position = P * V * M * vec4(position, 1.0);
}
